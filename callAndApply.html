//定义一个animal类

function Animal(){

this.name = 'Animal' ;
this.showName = function(){
    console.log(this.name);
}

}

//定义一个cat类

function Cat(){

this.name = 'Cat';

}

//创建两个类对象

var ani = new Animal();

var cat1 = new Cat();

//通过call或apply方法，将原本属于ani对象的showName方法交给当前对象cat1来使用

ani.showName.call(cat1);//输出：Cat
